# PR 3 — Polish (Analytics, Usage Indicator, Animations)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans.

**Goal:** Take the working product (PR 0+1+2) from "functional" to "feels polished and operationally sound." No new features — just observability, usability, and UX polish.

### Task 6: Email helper

**File:** `api/_lib/email.js`

```javascript
const { Resend } = require('resend');

let _resend = null;
function getResend() {
  if (_resend) return _resend;
  if (!process.env.RESEND_API_KEY) throw new Error('RESEND_API_KEY not configured');
  _resend = new Resend(process.env.RESEND_API_KEY);
  return _resend;
}

async function sendEmail(to, subject, body) {
  const resend = getResend();
  return resend.emails.send({
    from: 'Deal Hound Alerts <alerts@dealhound.pro>', // requires verified domain in Resend
    to: [to],
    subject,
    text: body
  });
}

module.exports = { sendEmail };
```

### Task 7: Cron endpoint

**File:** `api/cron/health-check.js`

```javascript
const { createClient } = require('@supabase/supabase-js');
const { sendEmail } = require('../_lib/email.js');

const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_SERVICE_KEY);

module.exports = async function handler(req, res) {
  // Vercel cron sends a User-Agent of vercel-cron — light auth check
  const auth = req.headers.authorization;
  if (auth !== `Bearer ${process.env.CRON_SECRET}`) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  const t2hr = new Date(Date.now() - 2 * 60 * 60 * 1000).toISOString();
  const { count: stale2h } = await supabase
    .from('deal_searches')
    .select('*', { count: 'exact', head: true })
    .eq('status', 'scanning')
    .lt('run_at', t2hr);

  if ((stale2h || 0) > 0) {
    const to = process.env.ALERT_EMAIL_TO || 'gideon@stonemontcap.com';
    await sendEmail(
      to,
      `[Deal Hound] ${stale2h} stale scans — Mac worker may be down`,
      `${stale2h} scans have been in 'scanning' status for over 2 hours.\n\n` +
      `Check the Mac mini:\n` +
      `1. Is it awake and online?\n` +
      `2. Is the dealhound heartbeat agent running?\n` +
      `3. Check ~/.claude/agents/dealhound/heartbeat.md for last update\n\n` +
      `Live health: https://dealhound.pro/api/admin/health?key=YOUR_KEY`
    );
  }

  return res.status(200).json({ stale_scans_2hr: stale2h || 0, alerted: (stale2h || 0) > 0 });
};
```

`formatRelative` helper in `dashboard/src/lib/utils.js`:

```javascript
export function formatRelative(isoDate) {
  const diffMs = Date.now() - new Date(isoDate).getTime();
  const diffHr = Math.floor(diffMs / (1000 * 60 * 60));
  if (diffHr < 1) return 'just now';
  if (diffHr === 1) return '1 hour ago';
  if (diffHr < 24) return `${diffHr} hours ago`;
  const diffDays = Math.floor(diffHr / 24);
  if (diffDays === 1) return 'yesterday';
  return `${diffDays} days ago`;
}
```
